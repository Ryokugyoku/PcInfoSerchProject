35 private List<WmicCpuProperty> getCpuUsagePerProcess() {
36 List<WmicCpuProperty> cpuPropertyList = new List<WmicCpuProperty>();
37 String output = getProcessRslt(0);
38 String title = "";
39 String value = "";
40 WmicCpuProperty cpuProperty = new WmicCpuProperty();
41
42 Boolean insertFlg = false;
43 int count = 1;
44 for (int i = 0; i < output.Length; i++)
45 {
46 char c = output[i];
47 switch (c)
48 {
49 case '\n':
50 if (count % 2 == 0)
51 {
52 insertFlg = true;
53 }
54 break;
55 case '\r':
56 break;
57 case '=':
58 count++;
59 break;
60 default:
61 if (count % 2 == 0 && count != 0)
62 {
63 value = value + c;
64 }
65 else
66 {
67 title = title + c;
68 }
69 break;
70 }
71 if (insertFlg && count % 6 == 0)
72 {
73 cpuProperty = setCpuProperty(cpuProperty, title, value);
74 cpuPropertyList.Add(cpuProperty);
75 cpuProperty = new WmicCpuProperty();
76 value = "";
77 title = "";
78 insertFlg = false;
79
80 count++;
81
82 }
83 if (insertFlg)
84 {
85 cpuProperty = setCpuProperty(cpuProperty, title, value);
86 value = "";
87 title = "";
88 insertFlg = false;
89 count++;
90 }
91 }
92
93 return cpuPropertyList;
