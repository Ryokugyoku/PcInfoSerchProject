115 private void Cpu(IHardware h)
116 {
117 Cpu cpu = new Cpu();
118 foreach (ISensor sensor in h.Sensors)
119 {
120 switch (sensor.SensorType)
121 {
122 case SensorType.Voltage:
123 if (sensor.Name.Contains("#"))
124 {
125 PcVoltages.Add(sensor.Value.GetValueOrDefault());
126 }
127 else
128 {
129 cpu.PackageVoltage = sensor.Value.GetValueOrDefault();
130 }
131 break;
132 case SensorType.Temperature:
133 if (sensor.Name.Contains("Package"))
134 {
135 cpu.PackageTemp = sensor.Value.GetValueOrDefault();
136 }
137 else if (sensor.Name.Contains("Max"))
138 {
139 //Cpu全体の観測された温度
140 cpu.MaxPackageTemp = sensor.Value.GetValueOrDefault();
141 }
142 else if (sensor.Name.Contains("TjMax"))
143 {
144 //核物理コアの観測された温度
145 }
146 else if (sensor.Name.Contains("#"))
147 {
148 PcTemperatures.Add(sensor.Value.GetValueOrDefault());
149 }
150 break;
151 case SensorType.Clock:
152 break;
153 case SensorType.Power:
154 break;
155 case SensorType.Load:
156 if (sensor.Name.Contains("Total"))
157 {
158 cpu.TotalCpuUsage = sensor.Value.GetValueOrDefault();
159 }
160 else
161 {
162 PcAllCoreProcess.Add(sensor.Value.GetValueOrDefault());
163 }
164 break;
165 }
166 }
167 GlobalObject.CpuMap.Add(date,cpu);
168 }
